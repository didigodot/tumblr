\item{``Integration by Differentiation'' }
   \begin{proof}
   \begin{IEEEeqnarray*}{+rCl+x*}
    \int x^{n}ln^{2}(x)dx\\
    \frac{\partial}{\partial n}x^{n}&=&x^{n}ln(x)\\
    \int x^{n}dx &=& \frac{1}{n+1} \\
    \int x^{n}ln(x)dx &=& -\frac{1}{(n+1)^{2}} \\
    \int x^{n}ln^{2}(x)dx &=& \frac{2}{(n+1)^{3}}\\
    \end{IEEEeqnarray*}
    \end{proof}
